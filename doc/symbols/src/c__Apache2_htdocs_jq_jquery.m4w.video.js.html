 15 (function($){
 16   /**
 17    * @constructor
 18    * @class ビデオ(videoタグ)を管理・操作
 19    * @param id ビデオに固有に付けられるID(videoタグのidと共通にすればなお良い)
 20    * @param body 対応するvideoタグのjQueryオブジェクト
 21    */
 22   Video = function(id, body){
 23     this.id = id;
 24     this.body = body;
 25     this.is_play = false;
 26     this.is_pause = false;
 27   };
 28 
 29   /**
 30    * ビデオファイルを読み込む<br>videoタグを作り、再生ができるときに指定した関数を呼び出す
 31    * @example {id: "hoge", src: ["/hoge.mp4","/hoge.avi"]}
 32    * @param options.id videoタグに一意につけられるID
 33    * @param options.src ファイルのURLを配列で指定(マルチブラウザ対応、ブラウザ内では指定したどれかを再生)
 34    * @param [options.width] ビデオの横幅<br>ピクセル単位<br>省略時は640
 35    * @param [options.height] ビデオの高さ<br>ピクセル単位<br>省略時は480
 36    * @param [options.left] ブラウザ(もしくは親ブロックからの右方向の位置<br>単位はピクセル<br>省略時は0(ブロックの左端)
 37    * @param [options.top] ブラウザ(もしくは親ブロックからの上方向の位置<br>単位はピクセル<br>省略時は0(ブロックの上端)
 38    * @param [options.z] ビデオの奥行き<br>CSSのz-indexと同じ<br>省略時は0
 39    * @param [options.body] ビデオを埋め込むブロック<br>jQueryオブジェクトを指定<br>省略時はbodyブロック($("body"))
 40    * @param [options.loop] ビデオを繰り返し再生するかを指定<br>省略時はfalse(繰り返さない)
 41    * @param [options.autofinish] 再生終了時、自動的にvideoタグを取り除くかを指定<br>省略時はfalse(取り除かない)
 42    * @return ビデオを再生できる状態に持ち込んでいるDeferredオブジェクト<br>コールバック関数の引数は、{type: "video", id: options.id, value: 生成したVideoオブジェクト}で示すオブジェクト
 43    */
 44   Video.load = function(options){
 45     var o = $.extend({
 46       id: this.id,
 47       width: 640,
 48       height: 480,
 49       left: 0,
 50       top: 0,
 51       z: 0,
 52       loop: false,
 53       autofinish: false,
 54       body: $("body")
 55     }, options);
 56     
 57     var $area = $("<video />");
 58     var defer = $.Deferred();
 59 
 60     $area.attr("id", o.id);
 61     $area.attr("preload", "metadata");
 62     $area.attr("loop", o.loop);
 63     $area.attr("width", o.width);
 64     $area.attr("height", o.height);
 65     $area.css("position", "absolute");
 66     $area.css("top", o.top);
 67     $area.css("left", o.left);
 68     $area.css("z-index", o.z);
 69     
 70     for(var i=0; i<o.src.length; i++){
 71       var $src = $("<source />");
 72       var url = o.src[i] + "?" + new Date().getTime();
 73       $src.attr("id", o.id+"_"+i);
 74       $src.attr("name", o.id);
 75       $src.attr("src", url);
 76       $src.appendTo($area);
 77     }
 78     $area.appendTo(o.body);
 79 
 80     $area[0].addEventListener("loadedmetadata", (function(){
 81       var d = defer;
 82       var video_id = o.id;
 83       var body = $area;
 84       return function(){ d.resolve({type: "video", id: video_id, value: new Video(video_id, body)}); };
 85     }).bind(this)());
 86 
 87     if(o.autofinish == true){
 88       $area[0].addEventListener("ended", (function(){
 89           var d = defer;
 90           var video_id = o.id;
 91           var video = $area[0];
 92           return function(){
 93             video.finish();
 94             d.resolve(video_id);
 95           };
 96       }).bind(this)());
 97     }
 98 
 99     return defer.promise();
100   };

102   /**
103    * ビデオを表示する
104    */
105   Video.prototype.show = function(){
106     $("video#"+this.id)[0].show();
107   };
108 
109   /**
110    * ビデオを非表示にする
111    */
112   Video.prototype.hide = function(){
113     $("video#"+this.id)[0].hide();
114   };
115 
116   /**
117    * ビデオの奥行きを設定する
118    */
119   Video.prototype.z = function(z){
120     var $v = $("video#"+this.id);
121     if(!z){ return $v.css("z-index"); }
122     $v.css("z-index", parseInt(z));
123   };
124 
125   /**
126    * ロードした音声の長さを返す<br>単位：秒
127    */
128   Video.prototype.length = function(){
129     return $("video#"+this.id)[0].duuration;
130   };
131 
132   /**
133    * 再生している位置を返す<br>単位：秒
134    */
135   Video.prototype.pos = function(){
136     return $("video#"+this.id).currentTime;
137   };
138 
139   /**
140    * 音声のボリューム(0.0(無音)～1.0(最大))を返す
141    */
142   Video.prototype.volume = function(){
143     return $("video#"+this.id)[0].volume;
144   };
145 
146   /**
147    * 音声のボリューム(0.0(無音)～1.0(最大))を設定する
148    */
149   Video.prototype.set_volume = function(vol){
150     return $("video#"+this.id)[0].volume = vol;
151   };
152 
153   /**
154    * ビデオを最初から再生する
155    */
156   Video.prototype.play = function(){
157     var v = $("video#"+this.id)[0];
158     if(!this.is_pause && !v.ended){ v.pause(); }
159     v.play();
160     this.is_play = true;
161     this.is_pause = false;
162   };
163 
164   /**
165    * ビデオの再生を停止する<br>同時に、ビデオの再生位置を先頭に戻す
166    */
167   Video.prototype.stop = function(){
168     var v = $("video#"+this.id)[0];
169     if(v.ended){ return; }
170     if(!this.is_pause){ v.pause(); }
171     v.currentTime = 0;
172     this.is_play = false;
173     this.is_pause = false;
174   };
175 
176   /**
177    * ビデオの再生を停止する<br>同時に、ビデオの再生位置はそのまま
178    */
179   Video.prototype.pause = function(){
180     var v = $("video#"+this.id)[0];
181     if(this.is_pause || v.ended){ return; }
182     v.pause();
183     this.is_play = false;
184     this.is_pause = true;
185   };
186 
187 
188   /**
189    * 停止時の再生位置からビデオを再生する
190    */
191   Video.prototype.resume = function(){
192     $("video#"+this.id)[0].play();
193     this.is_play = true;
194     this.is_pause = false;
195   };
196 
197   /**
198    * videoタグを取り除く
199    */
200   Video.prototype.remove = function(){
201     $("video#"+this.id)[0].remove();
202   };
203 
204   /**
205    * 再生を停止して、videoタグを取り除く
206    */
207   Video.prototype.finish = function(){
208     this.pause();
209     this.hide();
210     this.remove();
211   };
212 
213   window.m4w = $.extend({Video: Video}, window.m4w);
214 
215   window.m4w.AssetsLoader.loaders.video = function(options){
216     return Video.load(options);
217   };
