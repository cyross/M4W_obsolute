 14 (function($){
 15   /**
 16    * @class フォームを管理するクラス
 17    */
 18   Form = function(){
 19   };
 20 
 21   /**
 22    * ボタン(inputタグ・type="button")を管理・操作
 23    * @param options.id ボタンに固有に付けられるID(inputタグのidと共通にすればなお良い)
 24    * @param [options.name] ボタンのな前(inputタグのnameと共通にすればなお良い)
 25    * @param [options.left] ボタン(もしくは親ブロックからの右方向の位置<br>単位はピクセル<br>省略時は0(ブロックの左端)
 26    * @param [options.top] ボタン(もしくは親ブロックからの上方向の位置<br>単位はピクセル<br>省略時は0(ブロックの上端)
 27    * @param [options.z] ボタンの奥行き<br>CSSのz-indexと同じ<br>省略時は1000
 28    * @param [options.body] ボタンを埋め込むブロック<br>jQueryオブジェクトを指定<br>省略時はbodyブロック($("body"))
 29    * @param [options.css] CSS値をオブジェクトで指定<br>省略時は{}
 30    * @param [options.attr] attr値をオブジェクトで指定<br>jQueryオブジェクトを指定<br>省略時は{}
 31    * @param [options.events] 登録したいイベントの関数群<br>"イベント名": イベントハンドラ本体で登録する<br>省略時は{}
 32    */
 33   Form.create_button = function(options){
 34     var o = $.extend({
 35       type: "button",
 36       position: "absolute",
 37       z: 0,
 38       left: 0,
 39       top: 0,
 40       body: $("body"),
 41       events: { }
 42     }, options);
 43 
 44     if(!("name" in o)){ o.name = o.id; }
 45 
 46     var $pb = $("<input />", o);
 47     $pb.css("position", o.position);
 48     $pb.css("z-index", o.z);
 49     $pb.css("left", o.left);
 50     $pb.css("top", o.top);
 51     $pb.unbind("click");
 52     $pb.appendTo(o.body);
 53 
 54     for(name in o.css){ $pb.css(name, o.css[name]); }
 55     for(name in o.attr){ $pb.attr(name, o.attr[name]); }
 56     for(name in o.events){
 57       $pb.unbind(name);
 58       $pb.bind(name, o.events[name]);
 59     }
 60 
 61     return $pb;
 62   }
 63 
 64   window.m4w = $.extend({Form: Form}, window.m4w);
 65 })(jQuery);
