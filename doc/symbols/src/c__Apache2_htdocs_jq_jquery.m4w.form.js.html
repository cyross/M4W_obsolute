 21   /**
 22    * (private)セットアップ<既定オプションの追加>
 23    */
 24   Form.init = function(options, append_options){
 25     var o = $.extend({
 26       position: "absolute",
 27       z: 0,
 28       left: 0,
 29       top: 0,
 30       body: $("body"),
 31       events: { },
 32       classes: []
 33     }, append_options);
 34     return $.extend(o, options);
 35   }

 37   /**
 38    * (private)セットアップ<基本情報の追加>
 39    */
 40   Form.setup = function($obj, o){
 41     $obj.css("position", o.position);
 42     $obj.css("z-index", o.z);
 43     $obj.css("left", o.left);
 44     $obj.css("top", o.top);
 45     $obj.appendTo(o.body);
 46 
 47     for(var i=0; i<o.classes.length; i++){ $obj.addClass(o.classes[i]); }
 48     for(name in o.css){ $obj.css(name, o.css[name]); }
 49     for(name in o.attr){ $obj.attr(name, o.attr[name]); }
 50     for(name in o.events){
 51       $obj.unbind(name);
 52       $obj.bind(name, o.events[name]);
 53     }
 54   }

 63    * @param [options.body] ボタンを埋め込むブロック<br>jQueryオブジェクトを指定<br>省略時はbodyブロック($("body"))
 64    * @param [options.css] CSS値をオブジェクトで指定<br>省略時は{}
 65    * @param [options.attr] attr値をオブジェクトで指定<br>jQueryオブジェクトを指定<br>省略時は{}
 66    * @param [options.events] 登録したいイベントの関数群<br>"イベント名": イベントハンドラ本体で登録する<br>省略時は{}
 67    * @param [options.classes] 登録したいクラス<br>クラス名の配列で登録する<br>省略時は[]

 69   Form.create_button = function(options){
 70     var o = Form.init(options, {type: "button"});
 71 
 72     if(!("name" in o)){ o.name = o.id; }
 73 
 74     var $pb = $("<input />", o);
 75     this.setup($pb, o);
 76 
 77     return $pb;
 78   }

 84    * @param [options.left] 画面(もしくは親ブロックからの右方向の位置<br>単位はピクセル<br>省略時は0(ブロックの左端)

 86    * @param [options.z] 奥行き<br>CSSのz-indexと同じ<br>省略時は1000
 87    * @param [options.body] 埋め込み先ブロック<br>jQueryオブジェクトを指定<br>省略時はbodyブロック($("body"))
 88    * @param [options.css] CSS値をオブジェクトで指定<br>省略時は{}
 89    * @param [options.attr] attr値をオブジェクトで指定<br>jQueryオブジェクトを指定<br>省略時は{}
 90    * @param [options.events] 登録したいイベントの関数群<br>"イベント名": イベントハンドラ本体で登録する<br>省略時は{}
 91    * @param [options.classes] 登録したいクラス<br>クラス名の配列で登録する<br>省略時は[]

 93   Form.create_selectbox = function(options){
 94     var o = Form.init(options, {options: []});
 95 
 96     if(!("name" in o)){ o.name = o.id; }
 97 
 98     var $select = $("<select />", o);
 99     this.setup($select, o);
100 
101     for(var i=0; i<o.options.length; i++){
102       var opt = $.extend({
103         value: "",
104         selected: false
105       }, o.options[i]);
106       if(!("text" in opt)){ opt["text"] = opt.value; }
107 
108       var $opt = $("<option />");
109       $opt.val(opt.value);
110       $opt.html(opt.text);
111       if(opt.selected){ $opt.attr("selected", "selected"); }
112       $opt.appendTo($select);
113     }
114 
115     return $select;
116   }

132     var o = Form.init(options, {});
133 
134     var $label = $("<span />", o);
135     this.setup($label, o);
136 
137     if(typeof(o.text) == "string"){
138       $label.html(o.text);
139     }
140     else{
141       o.text.appendTo($label);
142     }
143 
144     return $label;
