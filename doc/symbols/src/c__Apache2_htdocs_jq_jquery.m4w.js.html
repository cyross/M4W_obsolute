 93    * 隠蔽中の時は描画しない<br>

 95    * <ul>

103   Renderer.default_render = function(context){
104     if(!this.v){ return; }
105     context.save();
106     context.setTransform(this.m11,this.m12,this.m21,this.m22,this.x-this.dx,this.y-this.dy);
107     context.translate(this.dx,this.dy);
108     context.globalAlpha = this.a;
109     context.drawImage(this.image, this.sx, this.sy, this.sw, this.sh, 0, 0, this.dw, this.dh);
110     context.restore();
111   };

166   /**
167    * 画像などのアセットを読み込み、指定した関数をコールバックする<br>
168    * すべてのロードに成功したときはsuccess関数、一つでも失敗したときはfailed関数を呼ぶ
169    * success関数の引数は以下のキーを持つ辞書オブジェクト<br>
170    * <table>
171    * <tr><th>Key</th><th>Value</th></tr>
172    * <tr><td>image</td><td>Imageオブジェクトの辞書オブジェクト</tr>
173    * <tr><td>sprite</td><td>Spriteオブジェクトの辞書オブジェクト</tr>
174    * <tr><td>bgm</td><td>BGMのSoundオブジェクトの辞書オブジェクト</tr>
175    * <tr><td>se</td><td>効果音のSoundオブジェクトの辞書オブジェクト</td></tr>
176    * </table>
177    * failed関数の引数はdefer.when.failedメソッドの引数の配列が渡される<br>
178    * @example 実際のImage/Sprite/Soundを取得するときは以下の方法でアクセス可能<br>
179    * 例1)<br>
180    * function success(assets){<br>
181    *    var image = assets.image.(id);<br>
182    *    var sprite = assets.sprite.(id);<br>
183    *    var bgm = assets.bgm.(id);<br>
184    *    var se = assets.se.(id);<br>
185    * }<br>
186    * 例2)<br>
187    * function success(assets){<br>
188    *    var image = assets["image"]["id"];<br>
189    *    var sprite = assets["sprite"]["id"];<br>
190    *    var bgm = assets["bgm"]["id"];<br>
191    *    var se = assets["se"]["id"];<br>
192    * }
193    * @param options.assets アセット設定オブジェクトの配列
194    * @param options.assets[].type アセットの形式<br>画像は"image"、スプライトは"sprite"、音声は"sound"で示される
195    * @param options.assets[].その他 Image,Sprite,Soundクラスコンストラクタの引数と同じものを指定
196    * @param [options.success] ロードが全て完了したときに呼び出されるコールバック関数<br>引数はアセット管理オブジェクト<br>内容は後述
197    * @param [options.failed] ロードに失敗した時に呼び出されるコールバック関数<br>引数は失敗した時に渡される引数リスト
198    */
199   AssetsLoader.load = function(options){
200     var o = $.extend({
201       success: function(assets){ return assets; },
202       failed: function(args){ alert("error occured!"); return args; }
203     }, options);
204 
205     var assets = {}
206 
207     for(var i=0; i<o.assets.length; i++){
208       var asset = o.assets[i]
209       assets[asset.id] = asset;
210     }
211 
212     var exec_assets = [];
213     for(var asset_id in assets){
214       var asset = assets[asset_id];
215       exec_assets.push((AssetsLoader.loaders[asset.type])(asset));
216     }
217     $.when.apply(this, exec_assets).then(function(){
218       var loaded_assets = {
219       };
220       for(var i=0; i<arguments.length; i++){
221         var asset = arguments[i];
222         if (!(asset.type in loaded_assets)){ loaded_assets[asset.type] = {}; }
223         loaded_assets[asset.type][asset.id] = asset.value;
224       }
225       o.success(loaded_assets);
226     }).fail(function(){
227       o.failed(arguments);
228     });
229   };
